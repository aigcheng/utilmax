  const debounce = (fn: { apply: (arg0: any, arg1: any) => void }, t: number) => {
    let timeId: any = null
    const delay = t || 500
    return function (this: any, ...args: any) {
      if (timeId) {
        clearTimeout(timeId)
      }
      timeId = setTimeout(() => {
        timeId = null
        fn.apply(this, args)
      }, delay)
    }
  }

export default debounce
